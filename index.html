// conversations.ts - Logic for handling chat history and messages export interface Message { role: 'user' | 'ai'; content: string; timestamp: number; } export interface ChatSession { id: string; title: string; messages: Message[]; lastUpdated: number; } const STORAGE_KEY = 'pocketGuru_chats'; /** * Saari purani chats ko LocalStorage se nikaalne ke liye */ export const getAllChats = (): Record => { const data = localStorage.getItem(STORAGE_KEY); return data ? JSON.parse(data) : {}; }; /** * Ek nayi chat session create karne ke liye */ export const createNewChat = (firstMessage: string = ""): ChatSession => { const id = `chat_${Date.now()}`; const newChat: ChatSession = { id: id, title: firstMessage ? firstMessage.substring(0, 25) + "..." : "New Conversation", messages: [], lastUpdated: Date.now() }; saveChat(newChat); return newChat; }; /** * Specific chat mein naya message add karne ke liye */ export const addMessageToChat = (chatId: string, role: 'user' | 'ai', content: string): void => { const chats = getAllChats(); if (chats[chatId]) { chats[chatId].messages.push({ role, content, timestamp: Date.now() }); chats[chatId].lastUpdated = Date.now(); // Agar pehla message hai toh title update karo if (chats[chatId].messages.length === 1 && role === 'user') { chats[chatId].title = content.substring(0, 25) + "..."; } localStorage.setItem(STORAGE_KEY, JSON.stringify(chats)); } }; /** * Individual chat save karne ke liye helper function */ export const saveChat = (chat: ChatSession): void => { const chats = getAllChats(); chats[chat.id] = chat; localStorage.setItem(STORAGE_KEY, JSON.stringify(chats)); }; /** * Kisi chat ko delete karne ke liye */ export const deleteChat = (id: string): void => { const chats = getAllChats(); delete chats[id]; localStorage.setItem(STORAGE_KEY, JSON.stringify(chats)); };
